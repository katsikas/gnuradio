00041 /*!
00042  * \brief An abstracted constellation object
00043  * \ingroup digital
00044  *
00045  * The constellation objects hold the necessary information to pass
00046  * around constellation information for modulators and
00047  * demodulators. These objects contain the mapping between the bits
00048  * and the constellation points used to represent them as well as
00049  * methods for slicing the symbol space. Various implementations are
00050  * possible for efficiency and ease of use.
00051  *
00052  * Standard constellations (BPSK, QPSK, QAM, etc) can be inherited
00053  * from this class and overloaded to perform optimized slicing and
00054  * constellation mappings.
00055  */
00056 class DIGITAL_API digital_constellation : public boost::enable_shared_from_this<digital_constellation>
00057 {
00058 public:
00059   digital_constellation (std::vector<gr_complex> constellation,
00060                          std::vector<unsigned int> pre_diff_code,
00061                          unsigned int rotational_symmetry,
00062                          unsigned int dimensionality);
00063   digital_constellation ();
00064 
00065   //! Returns the constellation points for a symbol value
00066   void map_to_points(unsigned int value, gr_complex *points);
00067   std::vector<gr_complex> map_to_points_v(unsigned int value);
00068 
00069   //! Returns the constellation point that matches best.
00070   virtual unsigned int decision_maker (const gr_complex *sample) = 0;
00071   //! Takes a vector rather than a pointer.  Better for SWIG wrapping.
00072   unsigned int decision_maker_v (std::vector<gr_complex> sample);
00073   //! Also calculates the phase error.
00074   unsigned int decision_maker_pe (const gr_complex *sample, float *phase_error);
00075   //! Calculates distance.
00076   unsigned int decision_maker_e (const gr_complex *sample, float *error);
00077   
00078   //! Calculates metrics for all points in the constellation.
00079   //! For use with the viterbi algorithm.
00080   virtual void calc_metric(const gr_complex *sample, float *metric, trellis_metric_type_t type);
00081   virtual void calc_euclidean_metric(const gr_complex *sample, float *metric);
00082   virtual void calc_hard_symbol_metric(const gr_complex *sample, float *metric);
00083   
00084   //! Returns the set of points in this constellation.
00085   std::vector<gr_complex> points() { return d_constellation;}
00086   //! Returns the vector of points in this constellation.
00087   //! Raise error if dimensionality is not one.
00088   std::vector<gr_complex> s_points();
00089   //! Returns a vector of vectors of points.
00090   std::vector<std::vector<gr_complex> > v_points();
00091   //! Whether to apply an encoding before doing differential encoding. (e.g. gray coding)
00092   bool apply_pre_diff_code() { return d_apply_pre_diff_code;}
00093   //! Whether to apply an encoding before doing differential encoding. (e.g. gray coding)
00094   void set_pre_diff_code(bool a) { d_apply_pre_diff_code = a;}
00095   //! Returns the encoding to apply before differential encoding.
00096   std::vector<unsigned int> pre_diff_code() { return d_pre_diff_code;}
00097   //! Returns the order of rotational symmetry.
00098   unsigned int rotational_symmetry() { return d_rotational_symmetry;}
00099   //! Returns the number of complex numbers in a single symbol.
00100   unsigned int dimensionality() {return d_dimensionality;}
00101 
00102   unsigned int bits_per_symbol () {
00103     return floor(log(double(d_constellation.size()))/d_dimensionality/log(2.0));
00104   }
00105   
00106   unsigned int arity () {
00107     return d_arity;
00108   }
00109 
00110   digital_constellation_sptr base() {
00111     return shared_from_this();
00112   }  
00113 
00114  protected:
00115 
00116   std::vector<gr_complex> d_constellation;
00117   std::vector<unsigned int> d_pre_diff_code; 
00118   bool d_apply_pre_diff_code;
00119   unsigned int d_rotational_symmetry;
00120   unsigned int d_dimensionality;
00121   unsigned int d_arity;
00122 
00123   float get_distance(unsigned int index, const gr_complex *sample);
00124   unsigned int get_closest_point(const gr_complex *sample);
00125   void calc_arity ();
00126 };
