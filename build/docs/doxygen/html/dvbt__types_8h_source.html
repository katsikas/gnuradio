00080         static unsigned int packets;
00081         // these three are mutually exclusive
00082         //     This is a regular data segment.
00083         static const int        flag01          = 0x0001;
00084         //       This is a field sync segment, for 1st half of a field.
00085         static const int        flag02          = 0x0002;
00086         //       This is a field sync segment, for 2nd half of a field.

00089         // This bit is set when Reed-Solomon decoding detects an error that it

00094         bool transport_error;
