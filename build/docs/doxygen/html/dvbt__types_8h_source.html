00113         // these three are mutually exclusive
00114         //     This is a regular data segment.
00115         static const int        fl_regular_seg          = 0x0001;
00116         //       This is a field sync segment, for 1st half of a field.

00121         // This bit is on ONLY when fl_regular_seg is set AND when this is
00122         // the first regular data segment AFTER a field sync segment.  This
00123         // segment causes various processing modules to reset.
00124         static const int        fl_first_regular_seg    = 0x0008;
00125 
00126         // which field are we in?
00127         static const int        fl_field2               = 0x0010;       // else field 1
00128 
00129         // This bit is set when Reed-Solomon decoding detects an error that it

00132         // then copied into the final Transport Stream packet so that MPEG
00133         // software can see that the 188-byte data segment has been corrupted.
00134         static const int        fl_transport_error      = 0x0020;
